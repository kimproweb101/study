```vue [@/layouts/default.vue]
<template>
  <div :style="pageContainerStyles">
    <RouterView />
  </div>  
</template>
<script setup>
import { computed } from 'vue';
import { useRoute } from 'vue-router';

const route = useRoute();
const pageContainerStyles = computed(() => ({
  maxWidth: route.meta?.width || '1080px',
  margin: '0 auto',
}));
</script>

```
